FUNCTION powerMod(b, e, m)
    x = 1
    WHILE e > 0
        IF e % 2 == 1
            x = (b * x) % m
        END IF
        b = (b * b) % m
        e = e / 2
    END WHILE
    RETURN x
END FUNCTION
