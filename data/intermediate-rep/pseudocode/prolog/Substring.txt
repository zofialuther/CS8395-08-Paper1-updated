substring_task(Str, N, M, Char, SubStr) 
    Span = sub_string(Str, N, M, _, _)
    ToEnd = sub_string(Str, N, _, 0, _)
    MinusLast = sub_string(Str, 0, _, 1, _)
    FromCharToMth = string_from_substring_to_m(Str, Char, M, _)
    FromSubToM = string_from_substring_to_m(Str, SubStr, M, _)
    writeln('from n to m ':Span)
    writeln('from n to end ': ToEnd)
    writeln('string minus last char ': MinusLast)
    writeln('form known char to m ': FromCharToMth)
    writeln('from known substring to m ': FromSubToM)

string_from_substring_to_m(String, Sub, M, FromSubToM) 
    Before = sub_string(String, _, _, _, Sub)
    FromSubToM = sub_string(String, Before, M, _, _)
